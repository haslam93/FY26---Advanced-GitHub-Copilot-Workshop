# Playwright UI Testing with GitHub Copilot

> **Goal:** Use Playwright MCP (from Module 03) + Copilot Chat to generate E2E tests for an enterprise service portal mock without manually writing a single locator.

---

## Prerequisites

- Playwright MCP connected in VS Code (see [Module 03 — Playwright MCP](../../03-mcp-samples/playwright-mcp/README.md))
- Node.js 20+
- Chromium installed: `npx playwright install chromium`

---

## Setup

```bash
cd 06-qa-testing/playwright-samples
npm install
npx playwright install chromium
```

### playwright.config.ts

```typescript
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests',
  fullyParallel: true,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: [['html', { open: 'never' }], ['list']],
  use: {
    // Points to the mock portal served locally
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
    screenshot: 'only-on-failure',
  },
  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
  ],
  // Auto-start the mock HTML server before tests
  webServer: {
    command: 'npx serve . -p 3000',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI,
  },
});
```

## Using Playwright MCP to Generate Locators

Instead of manually inspecting the DOM, delegate to Copilot + Playwright MCP:

```
Navigate to http://localhost:3000 and capture stable Playwright locators
for all interactive elements on the permit submission form.
Return them as a TypeScript object I can use in a page object model.
```

Playwright MCP will:
1. Open the browser to `http://localhost:3000`
2. Inspect the DOM
3. Return `getByRole`, `getByLabel`, and `getByTestId` locators — the most resilient selector types

## Copilot-Generated Test Examples

### permit-submission.spec.ts

```typescript
import { test, expect } from '@playwright/test';

test.describe('Permit Submission', () => {

  test.beforeEach(async ({ page }) => {
    await page.goto('/permits/new');
  });

  test('should submit a valid construction permit and show confirmation',
    async ({ page }) => {
      // Fill form using stable locators (generated by Playwright MCP)
      await page.getByLabel('Applicant Name').fill('Jane Doe');
      await page.getByLabel('Email').fill('jane.doe@example.com');
      await page.getByLabel('Permit Type').selectOption('CONSTRUCTION');
      await page.getByLabel('Region').selectOption('TORONTO');
      await page.getByLabel('Project Description').fill('Office renovation at 100 Queen St W');

      // Submit
      await page.getByRole('button', { name: 'Submit Application' }).click();

      // Assert confirmation
      await expect(page.getByRole('heading', { name: 'Application Submitted' }))
        .toBeVisible();
      await expect(page.getByTestId('confirmation-number'))
        .toHaveText(/^P-\d{6}$/);
    }
  );

  test('should show validation errors when required fields are empty',
    async ({ page }) => {
      await page.getByRole('button', { name: 'Submit Application' }).click();

      await expect(page.getByText('Applicant Name is required')).toBeVisible();
      await expect(page.getByText('Email is required')).toBeVisible();
      await expect(page.getByText('Permit Type is required')).toBeVisible();
    }
  );

  test('should reject invalid email format',
    async ({ page }) => {
      await page.getByLabel('Email').fill('not-an-email');
      await page.getByRole('button', { name: 'Submit Application' }).click();

      await expect(page.getByText('Please enter a valid email address'))
        .toBeVisible();
    }
  );

});
```

### permit-search.spec.ts

```typescript
import { test, expect } from '@playwright/test';

test.describe('Permit Search', () => {

  test('should find a permit by ID and display details', async ({ page }) => {
    await page.goto('/permits/search');
    await page.getByLabel('Permit ID').fill('P-000001');
    await page.getByRole('button', { name: 'Search' }).click();

    await expect(page.getByTestId('permit-status')).toBeVisible();
    await expect(page.getByTestId('permit-status')).toHaveText(/ACTIVE|PENDING|APPROVED/);
  });

  test('should show no-results message for unknown permit', async ({ page }) => {
    await page.goto('/permits/search');
    await page.getByLabel('Permit ID').fill('P-UNKNOWN');
    await page.getByRole('button', { name: 'Search' }).click();

    await expect(page.getByText('No permit found with this ID')).toBeVisible();
  });

});
```

### accessibility.spec.ts

```typescript
import { test, expect } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';

test.describe('Accessibility — WCAG 2.1 AA', () => {

  test('permit submission form has no critical accessibility violations',
    async ({ page }) => {
      await page.goto('/permits/new');

      const results = await new AxeBuilder({ page })
        .withTags(['wcag2a', 'wcag2aa', 'wcag21aa'])
        .analyze();

      // Report violations as readable test output
      expect(results.violations).toEqual([]);
    }
  );

  test('permit search page is keyboard navigable', async ({ page }) => {
    await page.goto('/permits/search');

    // Tab through all interactive elements
    await page.keyboard.press('Tab');
    await expect(page.getByLabel('Permit ID')).toBeFocused();

    await page.keyboard.press('Tab');
    await expect(page.getByRole('button', { name: 'Search' })).toBeFocused();
  });

});
```

## Running Tests

```bash
# Run all tests (headless)
npx playwright test

# Run with headed browser (see what's happening)
npx playwright test --headed

# Run a specific spec file
npx playwright test tests/permit-submission.spec.ts

# Run in debug mode (step through with DevTools)
npx playwright test --debug

# View HTML report
npx playwright show-report
```
